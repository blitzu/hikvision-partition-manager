**app/core/logging.py:**
```python
import json
import logging

class JsonFormatter(logging.Formatter):
    SKIP_ATTRS = frozenset(logging.LogRecord.__dict__) | {"message", "asctime", "args", "msg"}

    def format(self, record: logging.LogRecord) -> str:
        from app.middleware.logging import request_id_var
        log_obj = {
            "timestamp": self.formatTime(record, datefmt="%Y-%m-%dT%H:%M:%S"),
            "level": record.levelname,
            "logger": record.name,
            "message": record.getMessage(),
            "request_id": request_id_var.get(""),
        }
        for key, val in record.__dict__.items():
            if key not in self.SKIP_ATTRS:
                log_obj[key] = val
        # NVR-06: scrub password field from all log output
        log_obj.pop("password", None)
        if record.exc_info:
            log_obj["exception"] = self.formatException(record.exc_info)
        return json.dumps(log_obj, default=str)

def setup_logging(log_level: str = "INFO") -> None:
    handler = logging.StreamHandler()
    handler.setFormatter(JsonFormatter())
    logging.basicConfig(level=log_level.upper(), handlers=[handler], force=True)
```

**app/middleware/logging.py:**
```python
import time, uuid, logging
from contextvars import ContextVar
from starlette.middleware.base import BaseHTTPMiddleware
from starlette.requests import Request

request_id_var: ContextVar[str] = ContextVar("request_id", default="")
logger = logging.getLogger("http")

class RequestLoggingMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request: Request, call_next):
        rid = str(uuid.uuid4())
        token = request_id_var.set(rid)
        start = time.monotonic()
        response = await call_next(request)
        duration_ms = round((time.monotonic() - start) * 1000, 1)
        logger.info(
            "request",
            extra={
                "component": "http",
                "method": request.method,
                "path": request.url.path,
                "status_code": response.status_code,
                "duration_ms": duration_ms,
                "request_id": rid,
            },
        )
        request_id_var.reset(token)
        return response
```

**app/core/inflight.py:**
Use a simple module-level dict + lazy asyncio.Event pattern to avoid Python 3.10+ DeprecationWarning from creating asyncio.Event() at module import time:

```python
import asyncio

_state: dict = {"count": 0, "event": None}

def _get_event() -> asyncio.Event:
    """Lazily create asyncio.Event inside the running event loop."""
    if _state["event"] is None:
        _state["event"] = asyncio.Event()
        _state["event"].set()  # starts idle
    return _state["event"]

def increment() -> None:
    _state["count"] += 1
    _get_event().clear()

def decrement() -> None:
    _state["count"] -= 1
    if _state["count"] <= 0:
        _state["count"] = 0
        _get_event().set()

async def wait_drain(timeout: float = 30.0) -> int:
    """Wait for all in-flight calls to complete. Returns remaining count (0 on success)."""
    try:
        await asyncio.wait_for(_get_event().wait(), timeout=timeout)
        return 0
    except asyncio.TimeoutError:
        return _state["count"]

def reset() -> None:
    """Reset state — used in tests only."""
    _state["count"] = 0
    _state["event"] = None
```

Note: for test isolation, tests must call `inflight.reset()` in teardown or use monkeypatch.

from app.core import inflight

@asynccontextmanager
