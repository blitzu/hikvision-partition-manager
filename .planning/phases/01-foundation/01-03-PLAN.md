    Create app/isapi/client.py (Pattern 9 from RESEARCH.md — minimal client, extension point for Phase 2):
    ```python
    import httpx
    import xml.etree.ElementTree as ET

    class ISAPIClient:
        """
        Minimal ISAPI client for Phase 1.
        Phase 2 extends this class with: retry-on-timeout logic, detection type
        endpoints (motionDetection, LineDetection, FieldDetection, shelteralarm),
        and PUT methods for disarm/arm operations.
        """

        def __init__(self, host: str, port: int, username: str, password: str):
            self.base_url = f"http://{host}:{port}"
            self._auth = httpx.DigestAuth(username, password)
            # Per RESEARCH.md Pitfall 4: create new client per operation
            # (Hikvision firmware reuses nonces incorrectly with shared clients)
            self._client_kwargs = {
                "auth": self._auth,
                "verify": False,   # NVRs use self-signed certs (ISAPI-04)
                "timeout": httpx.Timeout(connect=5.0, read=10.0),  # ISAPI-02
            }

        async def get_device_info(self) -> dict:
            """GET /ISAPI/System/deviceInfo — used by connectivity test (NVR-03)."""
            async with httpx.AsyncClient(**self._client_kwargs) as client:
                resp = await client.get(f"{self.base_url}/ISAPI/System/deviceInfo")
                resp.raise_for_status()
                return self._parse_xml_to_dict(resp.text)

        async def get_camera_channels(self) -> list[dict]:
            """
            GET /ISAPI/System/Video/inputs/channels — used by camera sync (NVR-04).
            NOTE: This path is community-verified, not from official Hikvision docs.
            Some firmware variants use /ISAPI/ContentMgmt/StreamingProxy/channels.
            Verify against real device during integration testing.
            """
            async with httpx.AsyncClient(**self._client_kwargs) as client:
                resp = await client.get(
                    f"{self.base_url}/ISAPI/System/Video/inputs/channels"
                )
                resp.raise_for_status()
                return self._parse_channel_list(resp.text)

        def _parse_xml_to_dict(self, xml_text: str) -> dict:
            """Strip XML namespace and return tag->text dict."""
            root = ET.fromstring(xml_text)
            return {child.tag.split("}")[-1]: child.text for child in root}

        def _parse_channel_list(self, xml_text: str) -> list[dict]:
            """Parse VideoInputChannel elements from channel list XML."""
            root = ET.fromstring(xml_text)
            channels = []
            for ch in root.iter():
                if ch.tag.endswith("VideoInputChannel"):
                    id_el = ch.find(".//{*}id")
                    name_el = ch.find(".//{*}inputPort") or ch.find(".//{*}name")
                    if id_el is not None:
                        channels.append({
                            "channel_no": int(id_el.text),
                            "name": name_el.text if name_el is not None else f"Channel {id_el.text}",
                        })
            return channels
    ```

    Add connectivity test endpoint to app/nvrs/routes.py:
    ```python
    @router.get("/{nvr_id}/test")
    async def test_connectivity(
        nvr_id: uuid.UUID,
        db: Annotated[AsyncSession, Depends(get_db)],
    ) -> APIResponse[dict]:
        nvr = await db.get(NVRDevice, nvr_id)
        if not nvr:
            return APIResponse(success=False, error="NVR not found")

        # NVR-06: decrypt only within this function scope; never log or format into strings
        password = decrypt_password(nvr.password_encrypted)
        client_isapi = ISAPIClient(nvr.ip_address, nvr.port, nvr.username, password)

        try:
            device_info = await client_isapi.get_device_info()
            # NVR-05: update last_seen_at and status on every successful ISAPI contact
            nvr.last_seen_at = datetime.now(UTC)
            nvr.status = "online"
            await db.commit()
            return APIResponse(success=True, data=device_info)
        except Exception as exc:
            # NVR-06: log only class name — never format password or str(exc) which may contain URL with auth
            nvr.status = "offline"
            await db.commit()
            # NVR-03: always HTTP 200, success=false envelope on connectivity failure
            return APIResponse(
                success=False,
                data=None,
                error=f"NVR unreachable: {type(exc).__name__}"
            )
    ```

         ```python
         from sqlalchemy.dialects.postgresql import insert as pg_insert
         ```
