<!-- Key API contracts consumed by the UI. No codebase exploration needed. -->

GET /api/dashboard → APIResponse[DashboardResponse]
DashboardResponse:
  partitions: List[DashboardPartitionEntry]
  total: int
  active_count: int

DashboardPartitionEntry fields:
  id: UUID
  name: str
  description: str | None
  state: str | None          # "armed" | "disarmed" | "partial" | "error"
  disarmed_minutes: float | None
  overdue: bool
  scheduled_rearm_at: datetime | None
  last_changed_at: datetime | None
  last_changed_by: str | None

GET /api/partitions/{id}/state → APIResponse[PartitionStateRead]
PartitionStateRead fields:
  partition_id: UUID
  state: str | None
  last_changed_at: datetime | None
  last_changed_by: str | None
  scheduled_rearm_at: datetime | None
  error_detail: str | None
  cameras: List[CameraStateRead]

CameraStateRead fields:
  id: UUID
  channel_no: int
  name: str | None
  nvr_id: UUID
  detection_snapshot: dict | None   # None = camera currently armed
  disarmed_by_partitions: List[UUID]
  disarm_count: int

GET /api/partitions/{id}/audit?limit=20&offset=0 → APIResponse[PaginatedAuditLog]
PaginatedAuditLog:
  total: int
  limit: int
  offset: int
  items: List[AuditLogEntryRead]

AuditLogEntryRead fields:
  id: UUID
  partition_id: UUID
  action: str
  performed_by: str
  audit_metadata: dict | None
  created_at: datetime

POST /api/partitions/{id}/disarm body: { "disarmed_by": str, "reason": str | null }
POST /api/partitions/{id}/arm   body: { "armed_by": str }
Both return APIResponse[DisarmResponse | ArmResponse] with success bool.

GET /api/partitions/{id} → APIResponse[PartitionDetail]
PartitionDetail fields (extends PartitionRead):
  id, name, description, auto_rearm_minutes, alert_if_disarmed_minutes, state, created_at
  cameras: List[CameraRead]
